%% Extensions
\RequirePackage{multirow}

%% New columns types
\newcolumntype{R}[1]{>{\raggedleft\arraybackslash }b{#1}}
\newcolumntype{L}[1]{>{\raggedright\arraybackslash }b{#1}}
\newcolumntype{C}[1]{>{\centering\arraybackslash }b{#1}}

%% New Environments
\newenvironment{owntab}[1]{
	\begin{tabular}{|#1|}
		\hline
}{
		\hline
	\end{tabular}
}

\newenvironment{centertab}[2][]{
	\ifthenelse{\equal{#1}{}}{}{
		\addcontentsline{lot}{table}{#1}
		\refstepcounter{mytables}
	}
	\begin{center}
		\centering
		\begin{owntab}{#2}
}{
		\end{owntab}
	\end{center}
}
