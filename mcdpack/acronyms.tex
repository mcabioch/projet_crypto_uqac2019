\newcommandx{\createacronym}[8][4=, 5=, 6=, 7=, 8=]{%
	\ifthenelse{\equal{#4}{}}{%
		\newacronym[%
			shortplural={\textbf{\textit{%
				\glsindex[#4][#5]{#2}[#6][#2]s%
			}}},%
			longplural={#3s}%
		]{#1}{%
			\glsindex[#4][#5]{#2}[#6][#2]%
		}{#3}%
	}{%
		\newacronym[%
			shortplural={\textbf{\textit{%
				\glsindex[#6][#7]{#4}[#8][#2]%
			}}},%
			longplural={#5}%
		]{#1}{%
			\glsindex[#6][#7]{#2}[#8][#2]%
		}{#3}%
	}%
}
