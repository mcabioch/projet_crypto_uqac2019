%% New Commands
\newcommandx{\createacronym}[6][4=, 5=, 6=]{%
	\newacronym{#1}{%
		\glsindex[#4][#5]{#2}[#6][#2]%
	}{#3}%
}
\newcommandx{\createacronyms}[8][6=, 7=, 8=]{%
	\newacronym[%
		shortplural={\textbf{\textit{#4}}},%
		longplural={#5}%
	]{#1}{%
		\glsindex[#6][#7]{#2}[#8][#2]%
	}{#3}%
}
