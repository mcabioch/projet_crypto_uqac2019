%% New Commands
\newcommandx{\setsubsubsubsection}[2][2=]{%
	\phantomsection%
	\subsubsection*{#1}%
	\ifthenelse{\equal{#2}{}}{%
		\addcontentsline{toc}{subsubsection}{\tab#1}%
	}{%
		\addcontentsline{toc}{subsubsection}{\tab#2}%
	}%
	\phantomsection%
}

\newcommandx{\from}[3][2=, 3=]{%
	\begin{flushright}%
		\footnotesize{}\ttfamily{}%
			\fromcommandtext{} %
			\ifthenelse{\equal{#2}{}}{%
				#1%
			}{%
				\ifthenelse{\equal{#1}{}}{}{%
					#1 %
				}%
				\mail[#2]{#3}%
			}%
	\end{flushright}
}

%% Re-New Commands
\newcommand{\partstexts}{}

\if@numpart%
	\if@haspart
		\renewcommand{\thepart}{\Roman{part}}
		\renewcommand{\thesection}{\arabic{chapter}}
	\else
		\renewcommand{\thepart}{\arabic{part}}
		\renewcommand{\thesection}{\thepart{}.\arabic{chapter}}
	\fi

	\renewcommand{\thesubsection}{\thesection{}.\alph{section}}
	\renewcommand{\thesubsubsection}{\thesubsection{}-\roman{subsection}}
\fi%

\renewcommandx{\setpart}[2][2=]{%
	\renewcommand{\actualdocpart}{#1}%
%
	\stepcounter{part}%
	\if@numpart%
		\renewcommand{\partstexts}{\thepart{} }%
	\else%
		\renewcommand{\partstexts}{}%
	\fi%
%
	\phantomsection%
	\if@haspart%
		\part*{\partstexts{}#1}%
	\else%
		\section*{\partstexts{}#1}%
	\fi%
	\ifthenelse{\equal{#2}{}}{%
		\addcontentsline{toc}{part}{\partstexts{}#1}%
	}{%
		\addcontentsline{toc}{part}{\partstexts{}#2}%
	}%
	\phantomsection%
}

	\stepcounter{chapter}%
	\if@numpart%

%
	\phantomsection%

	\fi%

	\phantomsection%

		\subsubsection*{\partstexts{}#1}%
	\else
		\subsubsubsection*{\partstexts{}#1}%
	\fi
	\ifthenelse{\equal{#2}{}}{%
		\addcontentsline{toc}{subsection}{\tab{}\partstexts{}#1}%
	}{%

	}%

}
