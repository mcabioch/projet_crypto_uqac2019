%% New Commands
\newcommandx{\setpart}[2][2=]{%
	\renewcommand{\actualdocpart}{#1}
	\part*{#1}%
	\ifthenelse{\equal{#2}{}}{%
		\addcontentsline{toc}{part}{#1}%
		\phantomsection%
	}{%
		\addcontentsline{toc}{part}{#2}%
		\phantomsection%
	}%
}
