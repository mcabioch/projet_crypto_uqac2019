%% Re New Commands
\let\lastincludegraphics\includegraphics
\renewcommandx{\includegraphics}[3][1=, 2=]{%
	\ifthenelse{\equal{#2}{}}{}{%
		\refstepcounter{figure}%
		\refstepcounter{myfigures}%
%
		\ifthenelse{\equal{#2}{auto}}{%
			\addcontentsline{lof}{figure}{Figure~\thefigure}%
		}{%
			\addcontentsline{lof}{figure}{#2}%
		}%
	}%
	\lastincludegraphics[#1]{\import@path/#3}%
}

\newcommandx{\centerincludegraphics}[3][1=, 2=]{%
	\begin{center}
		\includegraphics[#1][#2]{#3}
	\end{center}%
}

\let\lasturl\url
\renewcommandx{\url}[1]{%
	\english%
	\lasturl{#1}%
	\french%
}

\let\lastcite\cite
\renewcommandx{\cite}[1]{%
	\refstepcounter{mybibs}%
	\lastcite{#1}%
}
