%% New Environments
\newenvironment{graph2D}[1][]{
	\begin{tikzpicture}[baseline]
		\begin{axis}[
			#1
		]
}{
		\end{axis}
	\end{tikzpicture}
}

\newenvironment{form}{
	\shorthandoff{;}%
	\begin{Form}
}{
	\end{Form}
	\shorthandon{;}
}

\newenvironmentx{definitionbox}[3][3=\definitionlinkword{}, usedefault]{
	\begin{samepage}
		\begin{mdframed}[backgroundcolor=yellow!15]
		\ifthenelse{\equal{#2}{}}{%
			\textbf{#1} :%
		}{%
			\textbf{#1} {#3} '\textbf{#2}' :%
		}
}{
		\end{mdframed}
	\end{samepage}
}

\newenvironment{definition}[1][]{
	\begin{definitionbox}{\definitioncontent{}}{#1}[\definitionlinkword]
}{
	\end{definitionbox}
}

\newenvironment{theorem}[1][]{
	\begin{definitionbox}{\theoremcontent{}}{#1}[\theoremlinkword]
}{
	\end{definitionbox}
}

\newenvironment{corollary}[1][]{
	\begin{definitionbox}{\corollarycontent{}}{#1}[\corollarylinkword]
}{
	\end{definitionbox}
}

\newenvironment{answer}{
	\footnotesize%
	\color{blue}%
}{}

\newenvironment{correction}{
	\footnotesize%
	\color{darkgreen}%
}{}

\newenvironment{fixed-algorithm}{
	\begin{algorithm}[H]
}{
	\end{algorithm}
}
